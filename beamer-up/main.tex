\section{Getting Started}
% ===========================================

\begin{frame}{Using the Theme}
  Load the theme in your preamble, set your metadata, and compile from the checked-in
  \texttt{beamer-up/} directory:

  \begin{itemize}
    \item \texttt{\textbackslash usetheme\{UP\}} --- basic usage
    \item \texttt{\textbackslash usetheme[showlogo]\{UP\}} --- logo on every slide
    \item \texttt{\textbackslash usetheme[showlogo,english]\{UP\}} --- English UP logotype
    \item \texttt{\textbackslash usetheme[showlogo,filipino]\{UP\}} --- Filipino UP logotype
    \item \texttt{\textbackslash usetheme[nl]\{UP\}} --- legacy Dutch-logo compatibility path
  \end{itemize}

  \vskip 6pt
  Recommended compile path: \texttt{latexmk -pdf main.tex}.\\
  Fallback path: \texttt{pdflatex}, \texttt{biber}, \texttt{pdflatex}, \texttt{pdflatex}.

  \vskip 6pt
  For a fuller walkthrough, see the repository usage guide at
  \texttt{docs/usage-guide.md}.
\end{frame}

\begin{frame}{Theme Options at a Glance}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Flags} \footnotesize
        Pass as options to \texttt{\textbackslash usetheme[...]}\texttt{\{UP\}}:
        \texttt{showlogo}, \texttt{english}, \texttt{filipino}, \texttt{nl}, \texttt{nosectionpage}.
      \end{block}
      \vskip 6pt
      \begin{block}{Special Frames} \footnotesize
        \texttt{\textbackslash standoutframe\{...\}} for emphasis.\\
        \texttt{\textbackslash thankframe\{...\}\{...\}} to close.\\
        \texttt{\textbackslash sectionframe\{...\}} for a manual divider.
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth} \footnotesize
      \begin{block}{Conference Title Support}
        \texttt{\textbackslash multiauthor} activates the multi-author layout.
        Use \texttt{\textbackslash addinstitute\{n\}\{Name\}} to list affiliations
        and \texttt{\textbackslash venue\{...\}} for the event label.
      \end{block}
      \vskip 6pt
      \begin{exampleblock}{Public Contract}
        The supported public loader is \texttt{UP}. Any remaining \texttt{UU}
        naming is migration detail or implementation residue only.
      \end{exampleblock}
    \end{column}
  \end{columns}

  \vskip 10pt
  The \texttt{showlogo} flag is active in this demo, so the top-right identity mark
  remains visible on normal content slides.
\end{frame}

\begin{frame}{Common Academic Starting Points}
  \begin{columns}[T]

\section{Defense and Class Report Patterns}
% ===========================================

\begin{frame}{Thesis Defense Narrative}
  A common defense structure can fit the template without custom macros:

  \begin{enumerate}
    \item Problem statement and scope
    \item Related work or institutional context
    \item Method or system design
    \item Results, limitations, and next steps
    \item Contribution summary and committee discussion prompts
  \end{enumerate}

  \vskip 8pt
  Use automatic section pages for the major milestones and a manual
  \texttt{\textbackslash sectionframe\{...\}} when you need an explicit pivot slide.
\end{frame}

\begin{frame}{Class Report Slide Pattern}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{block}{Recommended rhythm}
        \begin{itemize}
          \item one headline claim per slide
          \item short bullets instead of paragraph walls
          \item figures or tables when comparison matters
          \item one takeaway at the bottom when the result is easy to miss
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{exampleblock}{Example takeaway}
        For weekly reports or laboratory updates, a simple title, three bullets, and one
        chart is often stronger than a dense text slide.
      \end{exampleblock}
      \vskip 6pt
      \begin{alertblock}{Migration reminder}
        If your older deck still uses \texttt{\textbackslash usetheme\{UU\}}, migrate to
        \texttt{\textbackslash usetheme\{UP\}} before reusing this template.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\section{Colour Palette and Typography}
% ===========================================

\begin{frame}{UP Institutional Colours}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Core Palette}
        \textcolor{UPMaroon}{\rule{1em}{1em}}~Maroon \texttt{\#8E1537}\\[3pt]
        \textcolor{UPForestGreen}{\rule{1em}{1em}}~Forest green \texttt{\#005740}\\[3pt]
        \textcolor{UPYellow}{\rule{1em}{1em}}~Yellow \texttt{\#FFB81D}\\[3pt]
        \textcolor{UPBlack}{\rule{1em}{1em}}~Spot black \texttt{\#231F20}\\[3pt]
        \textcolor{white}{\rule{1em}{1em}}~White \texttt{\#FFFFFF}
      \end{alertblock}
    \end{column}
    \begin{column}{0.48\textwidth}
    \begin{exampleblock}{Usage Rules}
        \begin{itemize}
          \item Maroon carries hero surfaces and alerts.
          \item Forest green supports section structure and examples.
          \item Yellow stays an accent for rules, bullets, and emphasis.
          \item Spot black anchors body text and labels on yellow.
        \end{itemize}
    \end{exampleblock}
    \end{column}
  \end{columns}

  \vskip 8pt
  The public theme stays inside the UP four-colour system rather than the inherited
  multi-hue Utrecht palette.
\end{frame}

\begin{frame}{Typography and Body Copy}
  \begin{enumerate}
    \item \textbf{Optima} for titles and headings when locally available
    \item \textbf{Palatino} for body text on the validated baseline
    \item \textbf{Avenir/Helvetica} as documented heading fallbacks
  \end{enumerate}

  \vskip 10pt
  \begin{exampleblock}{Tiered Font Fallback}
    The theme selects a practical fallback chain rather than assuming every system has
    the same commercial fonts installed:
    \begin{itemize}
      \item \textbf{XeLaTeX/LuaLaTeX:} Optima -> Avenir -> Helvetica
      \item \textbf{pdfLaTeX:} Helvetica-style sans fallback with Palatino body text
    \end{itemize}
  \end{exampleblock}

  \vskip 6pt
  \footnotesize
  \textcolor{UPBlack80}{Compile with \texttt{xelatex} or \texttt{lualatex} for Optima when it exists on your system.}
\end{frame}

\begin{frame}{Lists and Blocks for Class Reports}
  Itemize now uses a single yellow marker family across levels:

  \begin{itemize}
    \item First-level items use \textcolor{UPYellow}{yellow} squares
    \begin{itemize}
      \item Second-level items stay yellow and shrink with depth
      \begin{itemize}
        \item Third-level items keep the same accent hue for consistency
      \end{itemize}
    \end{itemize}
    \item Items can contain \textbf{bold}, \emph{italic}, or \alert{alerted} text
    \item Keep bullets concise and let spacing carry hierarchy
  \end{itemize}

  \vskip 8pt
  Enumerate uses dark labels on yellow chips:
  \begin{enumerate}
    \item Background or objective
    \item Method or implementation note
    \item Result or recommendation
  \end{enumerate}
\end{frame}

\begin{frame}{Method Snapshot}
  The theme supports mathematical typesetting for technical defenses and research talks
  \cite{tai1994mathematical}:

$$
\int_{a}^{b} f(x)\,dx \approx
\frac{h}{2}\left[f(x_0) + 2\sum_{i=1}^{n-1} f(x_i) + f(x_n)\right]
$$

  \begin{alertblock}{Note}
    This is an alert block, suitable for assumptions, constraints, or a result that the
    audience should not miss during a defense or report.
  \end{alertblock}
\end{frame}

\begin{frame}[fragile]{Analysis Script Snippet}
The \texttt{listings} package is pre-configured with UP colours.
  Specify the language and write code verbatim:

  \begin{lstlisting}[language=Python]
def fibonacci(n):
    """Return the n-th Fibonacci number."""
    if n <= 1:
        return n
    a, b = 0, 1
    for _ in range(2, n + 1):
        a, b = b, a + b  # shift window
    return b

print(fibonacci(10))  # 55
  \end{lstlisting}
\end{frame}
